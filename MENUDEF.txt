AddOptionMenu "OptionsMenu"
{
	Submenu "Ghoul Spawns", "GhoulSpawnOptions"
}

Optionmenu "GhoulSpawnOptions"
{
	Title "Ghoul Spawn Options"
	StaticText "$GS_VERSION", GOLD
	Submenu "Ghoul Spawn Keybinds",		"GhoulSpawnKeybinds"
	Tooltip "Keybinds specific to Ghoul Spawn."
	Option "$GS_SPAWNER_NAME", 			GS_GhoulSpawn,		"OnOff"
	Tooltip "$GS_SPAWNER_INFO"
	Option "$GS_DESPAWN_NAME",			GS_Despawn,			"YesNo"
	Tooltip "$GS_DESPAWN_INFO"
	option "$GS_ALERT_NAME",				GS_AlertNoiseType,	"GSAlertType"
	Tooltip "$GS_ALERT_INFO"
	StaticText ""
	Option "$GS_TARGETING_NAME",			GS_Targeting,		"GSTarget"
	Tooltip "$GS_TARGETING_INFO"
	Option "$GS_GHOULSINFIGHT_NAME",		GS_GhoulsInfight,	"YesNo"
	Tooltip "$GS_GHOULSINFIGHT_INFO"
	Option "$GS_NOTARGETABLES_NAME",		GS_NoTargetables,	"YesNo"
	Tooltip "$GS_NOTARGETABLES_INFO"
}

OptionValue "GSAlertType"
{
	0, "None"
	1, "Sight Sounds"
	2, "Global"
}

OptionValue "GSTarget"
{
	-1,	"Players/Friends Only"
	0,	"Obey map rules"
	1,	"Ignore map rules"
}

Optionmenu "GhoulSpawnKeybinds"
{
	Title "Ghoul Spawn Keybinds"
	
	Control "Toggle Ghoul Spawning",	"NetEvent GS_ToggleSpawning"
